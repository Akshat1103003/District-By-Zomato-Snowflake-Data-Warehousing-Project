-- STEP 2.1: FIX CASE-SENSITIVE CATEGORY ISSUE
-- ================================================================
-- Fix the 1 record with lowercase 'movie' that got EVE instead of MOV
-- ================================================================

USE SCHEMA DISTRICT_ANALYTICS_DB.BRONZE_LAYER;

-- Fix lowercase 'movie' record
UPDATE DISTRICT_RAW_TABLE
SET EVENT_ID = REPLACE(EVENT_ID, 'EVE_', 'MOV_')
WHERE EVENT_CATEGORY = 'movie'
  AND EVENT_ID LIKE 'EVE_%';

-- Also standardize the case for these categories
UPDATE DISTRICT_RAW_TABLE
SET EVENT_CATEGORY = CASE 
    WHEN LOWER(EVENT_CATEGORY) = 'movie' THEN 'Movie'
    WHEN LOWER(EVENT_CATEGORY) = 'movies' THEN 'Movies'
    WHEN LOWER(EVENT_CATEGORY) = 'concert' THEN 'Concert'
    WHEN LOWER(EVENT_CATEGORY) = 'concerts' THEN 'Concerts'
    WHEN LOWER(EVENT_CATEGORY) = 'food festival' THEN 'Food Festival'
    ELSE EVENT_CATEGORY
END
WHERE EVENT_CATEGORY IN ('movie', 'movies', 'concert', 'concerts', 'food festival');

-- Verify fix
SELECT 
    EVENT_CATEGORY,
    SUBSTRING(EVENT_ID, 1, 3) AS prefix,
    COUNT(*) AS count
FROM DISTRICT_RAW_TABLE
WHERE EVENT_CATEGORY IN ('Movie', 'movie', 'Concert', 'concert', 'Food Festival', 'food festival')
GROUP BY EVENT_CATEGORY, prefix
ORDER BY EVENT_CATEGORY;
